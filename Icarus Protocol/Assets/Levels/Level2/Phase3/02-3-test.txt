distanceLeft = 253
distanceRight = 491
leftActive = False
rightActive = False
shieldPower = 100

def Distance(direction):
	global distanceLeft
	global distanceRight
	
	if direction == "left":	
		return distanceLeft
	elif direction == "right":
		return distanceRight
	else:
		return 0

def Activate(direction):
	global leftActive
	global rightActive
	global shieldPower

	if shieldPower > 0 and direction == "left":
		leftActive = True
		rightActive = False
	elif shieldPower > 0 and direction == "right":
		leftActive = False
		rightActive = True

def Deactivate():
	global leftActive
	global rightActive

	leftActive = False
	rightActive = False

def simulate():
	global leftActive
	global rightActive
	global distanceLeft
	global distanceRight

	try:
    		parent.execute_user_code()
	except Exception: 
    		pass

	if distanceLeft <= 0:
		if leftActive is True:
			distanceLeft = 999999
		else:
			parent.end_simulation(1)

	if distanceRight <= 0:
		if rightActive is True:
			parent.end_simulation(0)
		else:
			parent.end_simulation(2)

def simulate_tick():
	global leftActive
	global rightActive
	global shieldPower
	global distanceLeft
	global distanceRight

	if distanceLeft != 999999: distanceLeft -= 3
	distanceRight -= 3
	if leftActive or rightActive:
		shieldPower -= 5
		if shieldPower <= 0:
			leftActive = False
			rightActive = False
	return 0
