chargeState = 0
successful Firings = 0
timer = 10

def ChargeUp():
	global chargeState
	chargeState += 1

def ChargeAmount():
	global chargeState
	return chargeState

def Fire():
	global chargeState
	global successfulFirings

	if chargeState > 95 and chargeState < 105:
		successfulFirings += 1
		chargeState = 0
	else:
		parent.end_simulation(1)

def simulate():
	global successfulFirings

	try:
    		parent.execute_user_code()
	except Exception: 
    		pass

	if successfulFirings >= 8:
		parent.end_simulation(0)

def simulate_tick():
	global timer
	global chargeState

	timer -= 0.1
	if timer <= 0:
		parent.end_simulation(2)	

	if chargeState > 150:
		parent.end_simulation(3)

	return 0
